# NRLC AI Case Study System - Cron Examples
# Add these to your crontab: crontab -e

# Daily AI verification (2 AM UTC)
0 2 * * * cd /path/to/nrlc.ai && php scripts/ai_answer_crawler.php >> /path/to/nrlc.ai/logs/crawler.log 2>&1

# Daily case study updates (3 AM UTC, after crawler)
0 3 * * * cd /path/to/nrlc.ai && php bin/generate-case-study-updates.php >> /path/to/nrlc.ai/logs/updates.log 2>&1

# Weekly validation check (Sunday 4 AM UTC)
0 4 * * 0 cd /path/to/nrlc.ai && find data/case-studies -name "*.jsonld" -exec php bin/validate-case-study.php {} \; >> /path/to/nrlc.ai/logs/validation.log 2>&1

# Notes:
# - Replace /path/to/nrlc.ai with your actual path
# - Ensure PHP CLI is in PATH or use full path: /usr/bin/php
# - Ensure logs directory exists: mkdir -p logs
# - Set environment variables in crontab or use .env file
